概述 ​

ES6是ECMAScript 2015（ES2015）的缩写，是ECMAScript的第六版，是JavaScript语言的下一代标准，2015年6月正式发布。

ad-tip
- 时间：2015年6月

引入 ​

  • 外部引入

多个 script 引入最终在一个 script 中执行

js
<script type="text/javascript" src="scripts.js"></script>

语法 ​

变量 ​

var 可以在声明前访问，let 不可以

var 作用域正常，let 作用域只在最近的 {} 内

变量未赋值，默认为 undefined

js
var a = 1;
let b = 2;

函数 ​

函数声明

js
// 默认值
function add(a = 1, b = 2) {
    console.log(a + b);
}

// 可变参数
function add(...args) {
    console.log(args);
}

// 箭头函数
const add = (a, b) => {
    console.log(a + b);
}

// 回调函数
function add(a, b, ()=>{
    console.log(a + b);
})

// 匿名函数
(function () {
    console.log('Hello World!');
})()

数组 ​

  • 创建
js
var arr = [1, 2, 3];
var arr = new Array(1, 2, 3);
  • 遍历
js
for (var i = 0; i < arr.length; i++) {
    console.log(arr[i]);
}

    console.log(arr[i]);
}

arr.forEach(function (item,index) {
    console.log(item);
})
  • 添加
js
// 末尾与开头

arr.unshift(0);
arr.shift();

arr=[...arr1,...arr2]

var newarr = arr.filter(function (item,index) {
    return item > 2;
})

对象 ​

  • 创建
js
class Person {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
    say() {
        console.log(this.name + ' is ' + this.age + ' years old.');
    }
}

字符串 ​

字符以 UTF-16 编码，每个字符占两个字节。

js
# 字符串拼接
var str = 'Add ' + a + ' and ' + b;
var str = `Add ${a} and ${b}`;

# 截取前三位的字符串
var str = str.slice(0, 3);

var str = str.replace('Hello', 'Hi');
js
// 字符编码，可以处理 UTF-16
let a = str.codePointAt(0);

// 字符解码
let b = String.fromCodePoint(a);

js
var reg = /\d{4}-\d{2}-\d{2}/;
