C in CPU ​

以下部分内容来自阿布编程的书籍"CPU 眼中的 C 语言" 对于汇编与原语句的对应关系,使用这个网站可以很清晰的比较: https://godbolt.org/ GitHub:https://github.com/compiler-explorer/compiler-explorer

内存分配 ​

stack 栈，向下延伸 高地址

heap 堆，向上延伸，动态分配

.bss 未初始化的全局变量 .data 初始化的全局变量 .text 代码段，rip保存下一条指令地址 低地址

编译的代码中的地址是虚拟地址，CPU只知道虚拟地址，需要经过MMU映射为物理地址。 不同进程之间的地址是完全隔离的，虽然虚拟地址一样，但实际分配的是不同的物理地址。 同一进程的各个线程之间共享内存，使用同一个地址空间。

静态变量无论在何处声明，内存分配和全局变量一致。初始化为 0 的全局变量仍然被认为是未初始化的，保存在.bss区。

静态变量不会被链接器链接到其他文件，只在当前文件使用。

函数栈帧 ​

rbp指向当前函数栈帧的栈底，rsp指向当前函数栈帧的栈顶；进入函数时，先保存上一个函数的rbp，然后将rbp下调至rsp。

c
main:	    1        a           高地址
main:	    2        b
main:  0x 0401 105e  返回地址
main:  0x 8000 0030  main的rbp
add :	    1
add :	    2                    低地址
c
#include <stdlib.h>

int add(int a, int b){
    return a + b;
}

int main(){
    int a = 1;
    int b = 2;
    int c = add(a,b);
    return 0;
}
c
add:
		// 保存 rbp 并更新 rbp
        push    rbp
        mov     rbp, rsp
		// 保存参数
        mov     DWORD PTR [rbp-4], edi
        mov     DWORD PTR [rbp-8], esi

        mov     edx, DWORD PTR [rbp-4]
        mov     eax, DWORD PTR [rbp-8]
        add     eax, edx
		// 退栈，恢复原栈帧
        pop     rbp
        ret

main:
		// 保存 rbp 并更新 rbp
        push    rbp
        mov     rbp, rsp
		// 预留临时变量空间
        sub     rsp, 16
		// 临时变量赋值
        mov     DWORD PTR [rbp-4], 1
        mov     DWORD PTR [rbp-8], 2
		// 传递调用函数参数
        mov     edx, DWORD PTR [rbp-8]
        mov     eax, DWORD PTR [rbp-4]
        mov     esi, edx
        mov     edi, eax
		// 先压入返回地址，再 goto 函数入口
        call    add
		// 保存返回值
        mov     DWORD PTR [rbp-12], eax
        mov     eax, 0
		// 退栈，恢复原栈帧
        leave
        ret

函数传参 ​

所有传参都是值传递，指针和引用传递地址，都是数据。如果要修改传入的参数，应该传入该参数的内存地址，这样才能修改值；否则，传入的值会保存在新的栈帧中，不是原来的值。

rdi，rsi，rdx，rcx，r8，r9xmm0至xmm7浮点参数

c
#include <stdlib.h>

void clear(int** p){
    free(*p);
    *p = 0;
}

int main(){
    int * p = (int*)malloc(sizeof(int));
    clear(&p);
    return 0;
}
c
clear:
        push    rbp
        mov     rbp, rsp
        sub     rsp, 16
        // free
        mov     QWORD PTR [rbp-8], rdi
        mov     rax, QWORD PTR [rbp-8]
        mov     rax, QWORD PTR [rax]
        mov     rdi, rax
        call    free
       
        mov     rax, QWORD PTR [rbp-8]
        mov     QWORD PTR [rax], 0
        nop
        leave
        ret

main:
        push    rbp
        mov     rbp, rsp
        sub     rsp, 16
        // malloc
        mov     edi, 4
        call    malloc
        mov     QWORD PTR [rbp-8], rax
        // clear
        lea     rax, [rbp-8]
        mov     rdi, rax
        call    clear
       
        mov     eax, 0
        leave
        ret

volatile ​

volatile可以阻止编译器过度优化变量，使得每次都从内存中读取变量，而不是作为常数优化。

c
int a = 0;

int main(){
    while(a > 1){
   
    }
    return 0;
}
  • 未使用优化
c
a:
        .zero   4

main:
        push    rbp
        mov     rbp, rsp
        nop

.L2:
        mov     eax, DWORD PTR a[rip]
        cmp     eax, 1
        jg      .L2
        mov     eax, 0
        pop     rbp
        ret
  • 使用-O2优化，直接消除while循环
c
a:
        .zero   4

main:
        push    rbp
        mov     rbp, rsp
        mov     eax, 1
        ret

加入volatile可以避免此类情况：

c
volatile int a = 0;

int main(){
    while(a > 1){
   
    }
    return 0;
}

const ​

常量的不可修改在于其分配的内存页为只读的，但在函数内部声明的栈常量是可以被修改的。

c
int func(){
	const int a = 1;
	*(int*)(&a) = 2;
	return a;
}
c
const int tol = 100;
// 错误，常量只是只读的变量，不可以用于数组长度
int a[tol];
c
#define tol (100)
// 正确，宏会在编译时自动展开
int a[tol];

assert ​

只在表达式为假时阻止程序运行。

c
#include <assert.h>

int main(){
	assert(str[i] == '\0');
	assert(0);
}

printf ​

va_arg会将ap自动移动至下一个位置。

c
#include <stdarg.h>
#include <stdio.h>
#include <assert.h>

void tiny_printf(char* format, ...){
	int i;
	va_list ap;

	va_start(ap, format);
	for(i = 0; format[i] != '\0'; i++){
		switch(format[i]){
			case 's':
				printf("%s", va_arg(ap, char*));
				break;
			case 'd':
				printf("%d", va_arg(ap, int));
				break;
			default:
				assert(0);
		}
	}
	va_end(ap);
	putchar('\n');
}

int main(){
	tiny_printf("sdd", "result..", 3 , 5);
	return 0;
}
c
void debug_write(char* fmt, ...){
	va_list ap;
	va_start(ap, fmt);
	vfprintf(stderr, fmt, ap);
	va_end(ap);
}
c
#ifdef DEBUG
	#define DEBUG_WRITE(arg) debug_write arg
#else
	#define DEBUG_WRITE(arg)
#endif

// 使用双括号
DEBUG_WRITE(("hoge...%d\n", hoge));
c
// C99
#ifdef DEBUG
	#define DEBUG_WRITE(...) debug_write (__VA_ARGS__)
#else
	#define DEBUG_WRITE(...)
#endif

DEBUG_WRITE("hoge...%d\n", hoge);
c
#ifdef DEBUG
	#define DEBUG_WRITE(...) \
	(debug_write ("%s:%s:%d", __FILE__, __func__, __LINE__),\
	debug_write (__VA_ARGS__))
#else
	#define DEBUG_WRITE(...)
#endif
